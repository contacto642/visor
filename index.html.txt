  <meta charset="utf-8" />

  <!-- Leaflet -->
  <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css"/>
  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <!-- Plugin KML -->
  <script src="https://unpkg.com/leaflet-omnivore/leaflet-omnivore.min.js"></script>

  <style>
    body { margin: 0; }
    #map { width: 100%; height: 100vh; }

    .leaflet-control-layers {
      font-size: 14px;
    }
  </style>

<body>

<div id="map"></div>

<script>
  // Crear mapa
  var map = L.map('map').setView([8.5, -72.5], 10);

  // Mapa base
  var base = L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap'
  }).addTo(map);

  // === CAPAS KML ===
  var conflictos = omnivore.kml('conflictos.kml');
  var inundaciones = omnivore.kml('inundaciones.kml');
  var coberturas = omnivore.kml('coberturas.kml');
  var drenajesDobles = omnivore.kml('drenajes_dobles.kml');
  var drenajesSencillos = omnivore.kml('drenajes_sencillos.kml');
  var limite = omnivore.kml('limite.kml');
  var redVial = omnivore.kml('red_vial.kml');

  // Agregar capas al mapa
  conflictos.addTo(map);
  inundaciones.addTo(map);
  coberturas.addTo(map);
  drenajesDobles.addTo(map);
  drenajesSencillos.addTo(map);
  limite.addTo(map);
  redVial.addTo(map);

  // Ajustar vista automáticamente con el límite
  limite.on('ready', function() {
    map.fitBounds(limite.getBounds());
  });

  // Control de capas
  var overlayMaps = {
    "Conflictos de uso": conflictos,
    "Zonas inundables": inundaciones,
    "Coberturas": coberturas,
    "Drenajes dobles": drenajesDobles,
    "Drenajes sencillos": drenajesSencillos,
    "Límite del área": limite,
    "Red vial": redVial
  };

  L.control.layers(null, overlayMaps, {
    collapsed: false
  }).addTo(map);

</script>

</body>
